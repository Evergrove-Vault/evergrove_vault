```dataviewjs
// –ø–æ–ª—É—á–∞–µ–º course-id –∏–∑ —Ç–µ–∫—É—â–µ–π —Å—Ç—Ä–∞–Ω–∏—Ü—ã
let current = dv.current();
let courseId = current["course-id"];

if (!courseId) {
    dv.paragraph("‚ùå –£ —Ç–µ–∫—É—â–µ–π —Å—Ç—Ä–∞–Ω–∏—Ü—ã –Ω–µ—Ç course-id");
} else {
    // –∏—â–µ–º –∫–Ω–∏–≥–∏ —Å —ç—Ç–∏–º course-id
    let books = dv.pages()
        .where(p => p.file.tags && p.file.tags.includes("#book"))
        .where(p => p["course-id"] == courseId)
        .sort(p => p.file.name);  // —Å–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞ –ø–æ –∏–º–µ–Ω–∏ —Ñ–∞–π–ª–∞

    let table = [];

    for (let book of books) {
        // —É–±–∏—Ä–∞–µ–º –ø—Ä–µ—Ñ–∏–∫—Å book_
        let displayName = book.file.name.replace(/^book_/, "");

        // –∏–º—è —Å—Ç—Ä–∞–Ω–∏—Ü—ã –∫—É—Ä—Å–∞: "<course-id> –∫–Ω–∏–≥–∏ –ø–æ –ø—Ä–æ–≥—Ä–∞–º–º–µ"
        let programPage = `${courseId} –∫–Ω–∏–≥–∏ –ø–æ –ø—Ä–æ–≥—Ä–∞–º–º–µ`;

        // —Å—Å—ã–ª–∫–∞ –Ω–∞ –∫–Ω–∏–≥—É
        let bookLink = `[[${programPage}#${displayName}|${displayName}]]`;

        // —Å–æ–±–∏—Ä–∞–µ–º —Å—Å—ã–ª–∫–∏ MOC (inlinks + outlinks)
        let mocPages = dv.pages("#MOC")
            .where(moc => 
                moc.file.inlinks.some(l => l.path == book.file.path) ||
                moc.file.outlinks.some(l => l.path == book.file.path)
            );

        // —Å–æ–∑–¥–∞–µ–º –∫–ª–∏–∫–∞–±–µ–ª—å–Ω—ã–µ —Å—Å—ã–ª–∫–∏ –∏ —Å–æ—Ä—Ç–∏—Ä—É–µ–º –ø–æ –∏–º–µ–Ω–∏
        let mocLinks = mocPages.length > 0
            ? mocPages
                .sort(m => m.file.name)
                .map(m => `[[${m.file.path}|${m.file.name}]]`)
                .join(", ")
            : "‚ùå –ù–µ—Ç MOC";

        table.push([bookLink, mocLinks]);
    }

    // –≤—ã–≤–æ–¥
    if (table.length > 0) {
        dv.table(["üìñ –ö–Ω–∏–≥–∞", "üìë MOC —Å–≤—è–∑–∞–Ω–Ω—ã–µ"], table);
    } else {
        dv.paragraph(`‚ùå –ù–µ—Ç –∫–Ω–∏–≥ —Å #book –¥–ª—è course-id: ${courseId}`);
    }
}
```
